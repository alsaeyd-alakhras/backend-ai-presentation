الـ HTTP Methods الأربعة
GET جلب البيانات — لا يغير شيء
// عرض صفحة المنتجات
Route::get('/products', function() {
  return view('products.index');
});
POST إنشاء بيانات جديدة
// حفظ منتج جديد
Route::post('/products', function() {
  // حفظ في قاعدة البيانات
});
PUT تحديث بيانات موجودة (كاملة)
// تحديث منتج بالكامل
Route::put('/products/{id}', function() {
  // تحديث في قاعدة البيانات
});
DELETE حذف بيانات
// حذف منتج
Route::delete('/products/{id}', function() {
  // حذف من قاعدة البيانات
});
💡 PATCH مثل PUT لكن يحدث جزء من البيانات فقط — سنستخدمه لاحقاً مع Eloquent

Route Parameters5 / 15
Route Parameters
✅ Required Parameter — إجباري
// {id} إجباري — يجب أن يُمرَّر
Route::get('/products/{id}',
 function($id) {
  return 'Product: ' . $id;
});

// SouqPro: عرض تفاصيل منتج
Route::get(
 '/products/{product}',
 [ProductController::class,
  'show']);
/products/5 → $id = 5
vs
🔵 Optional Parameter — اختياري
// {page?} اختياري — يعطيه قيمة افتراضية
Route::get('/products/{page?}',
 function($page = 1) {
  return 'Page: ' . $page;
});

// تحقق من قيمة المعامل
Route::get(
 '/products/{id}',
 ...)->where('id', '[0-9]+');
/products → $page = 1 (افتراضي)
في SouqPro سنستخدم Route Model Binding — لما تكتب Product $product داخل الـ Controller يقوم Laravel تلقائياً بما يشبه Product::findOrFail($id). سنشرحه بالتفصيل لاحقاً في محاضرة Eloquent، وهنا فقط نأخذه كمقدمة سريعة

Route Groups7 / 15
Route Groups — تجميع الـ Routes
prefix تجميع بـ URL مشترك كل الـ routes تبدأ بـ /admin
Route::prefix('admin')->group(function() {
  Route::get('/products', [AdminProductController::class, 'index']);
  Route::get('/orders', [AdminOrderController::class, 'index']);
}); // → /admin/products و /admin/orders
middleware حماية مجموعة Routes بـ Middleware يتحقق من تسجيل الدخول
Route::middleware(['auth', 'admin'])
  ->prefix('admin')
  ->name('admin.')
  ->group(function() {
    Route::resource('products', ProductController::class);
  }); // → admin.products.index, admin.products.show ...
💡 في SouqPro سيكون عندنا Route Group خاص بالـ Admin — يحتاج تسجيل دخول وصلاحية admin معاً

app/Http/Controllers/ProductController.php
class ProductController extends Controller {

 public function index() {
  $products = Product::latest()->paginate(12);
  return view('products.index',
   compact('products'));
 }

 public function show(Product $product) {
  // Route Model Binding ✨
  return view('products.show',
   compact('product'));
 }

 public function store(Request $request) {
  $request->validate(['name' => 'required']);
  Product::create($request->all());
  return redirect()->route('products.index');
 }
}
